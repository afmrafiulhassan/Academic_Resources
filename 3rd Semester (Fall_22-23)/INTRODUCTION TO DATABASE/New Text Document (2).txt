CREATE TABLE PHONE_NUMBER
(
 PID VARCHAR2(15) PRIMARY KEY,
 PHONE VARCHAR2(11) NOT NULL
)

DESCRIBE PHONE_NUMBER

CREATE TABLE ADDRESS
(
 AID VARCHAR2(15) PRIMARY KEY,
 H_NO NUMBER(6),	
 STREET VARCHAR2(15),
 CITY VARCHAR2(15)
)

DESCRIBE ADDRESS


CREATE TABLE ORGAN_BANK
(
   OBID VARCHAR2(15) PRIMARY KEY ,
   OB_NAME VARCHAR2(15),
   AID VARCHAR2(15),
   FOREIGN KEY (AID) REFERENCES ADDRESS (AID)
)

DESCRIBE ORGAN_BANK

CREATE TABLE ORGAN
(
   OID VARCHAR2(15) PRIMARY KEY,
   BLOOD_GROUP VARCHAR2(15),
   ORGAN_TYPE VARCHAR2(15),
   OS_DATE DATE,
   OBID VARCHAR2(15),
   FOREIGN KEY (OBID) REFERENCES ORGAN_BANK (OBID)
)

DESCRIBE ORGAN


CREATE TABLE DONOR
( 
  DID VARCHAR2(15) PRIMARY KEY, 
  D_NAME VARCHAR2(15),
  AGE VARCHAR2(5), 
  BLOOD_GROUP VARCHAR2(15), 
  GENDER VARCHAR2(15),
  OID VARCHAR2(15), PID VARCHAR2(15), AID VARCHAR2(15),
  FOREIGN KEY (OID) REFERENCES ORGAN (OID),
  FOREIGN KEY (PID) REFERENCES PHONE_NUMBER (PID),
  FOREIGN KEY (AID) REFERENCES ADDRESS (AID)
)
DESCRIBE DONOR


CREATE TABLE RECEPTIONIST
(
  RID VARCHAR2(15) PRIMARY KEY , 
  R_NAME VARCHAR2(15),
  OBID VARCHAR2(15),
  PID VARCHAR2(15),
  AID VARCHAR2(15),
  FOREIGN KEY (OBID) REFERENCES ORGAN_BANK (OBID),
  FOREIGN KEY (PID) REFERENCES PHONE_NUMBER (PID),
  FOREIGN KEY (AID) REFERENCES ADDRESS (AID)
)

DESCRIBE RECEPTIONIST


CREATE TABLE HOSPITAL
(
  HID VARCHAR2(15) PRIMARY KEY,
  H_NAME VARCHAR2(15),
  NO_OF_ORDER VARCHAR2(15),
  ISSUE_DATE VARCHAR2(15),
  OBID VARCHAR2(15), PID VARCHAR2(15), AID VARCHAR2(15),
  FOREIGN KEY (OBID) REFERENCES ORGAN_BANK (OBID),
  FOREIGN KEY (PID) REFERENCES PHONE_NUMBER (PID),
  FOREIGN KEY (AID) REFERENCES ADDRESS (AID)
)
DESCRIBE HOSPITAL



CREATE TABLE MANAGER
(
  MID VARCHAR2(15) PRIMARY KEY,
  M_NAME VARCHAR2(15),
  EMAIL_ID VARCHAR2(30),
  OBID VARCHAR2(15),
  PID VARCHAR2(15),
  AID VARCHAR2(15),
  FOREIGN KEY (OBID) REFERENCES ORGAN_BANK (OBID),
  FOREIGN KEY (PID) REFERENCES PHONE_NUMBER (PID),
  FOREIGN KEY (AID) REFERENCES ADDRESS (AID)
)
DESCRIBE MANAGER
